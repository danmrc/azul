library(gganimate)

m = 10000000
n = 100
amostras = vector()

for(i in 1:m) {
  set.seed(i)
  
  amostra = rbinom(n = n,
                   size = 1,
                   prob = .5)
  
  amostras[i] = mean(amostra)
    
}


summary(amostras)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##    0.23    0.47    0.50    0.50    0.53    0.75
ngrupos = 20
grupo = rep(seq(1, ngrupos), times = m/ngrupos)
grupo = sort(grupo)

base = data.frame(dados = amostras,
                  grupo = grupo)
